### Load modules
module load gcc/9.2.0 bedtools/2.29.2

### Working & temp directory
wd="/project/berglandlab/connor"
out=${wd}/control_cnvs

# Total CNV regions
cnv=${wd}/cnvs/CNV_all_genes.bed

# Positive controls - chitinases
chitin=${out}/chitin.genes

# Find overlap
bedtools intersect \
-a ${cnv} \
-b ${chitin} \
-wa | sort | uniq > \
${out}/sort.unique.chitin.cnv.hits
